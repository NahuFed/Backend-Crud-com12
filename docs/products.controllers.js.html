/**
 * Controladores relacionados con productos.
 * @module controllers/products
 */

/**
 * Tipos auxiliares para JSDoc — evitar usar import('...') en expresiones de tipo
 * @typedef {Object} ExpressRequest
 * @typedef {Object} ExpressResponse
 */

import Product from "../models/productsSchema.js";
/**
 * Obtiene todos los productos.
 * @async
 * @param {ExpressRequest} req - Objeto Request de Express.
 * @param {ExpressResponse} res - Objeto Response de Express.
 * @returns {Promise<void>}
 */
export const obtenerProductos = async (req, res) => {
  try {
    const products = await Product.find();
    res.status(200).json(products);
  } catch (error) {
    console.error(error);
    res.status(500).json({ 
      mensaje: "Error interno del servidor al obtener productos" 
    });
  }
};
/**
 * Obtiene un producto por su id.
 * @async
 * @param {ExpressRequest} req - Request. `req.params.id` es el id del producto.
 * @param {ExpressResponse} res - Response.
 * @returns {Promise<void>}
 */
export const obtenerUnProducto = async (req, res) => {
  try {
    const product = await Product.findById(req.params.id);
    if (!product) {
      return res.status(404).json({
        mensaje: "El producto no fue encontrado"
      });
    }
    res.status(200).json(product);
  } catch (error) {
    console.error(error);
    res.status(500).json({
      mensaje: "Error interno del servidor al obtener el producto"
    });
  }
};
/**
 * Crea un nuevo producto.
 * @async
 * @param {ExpressRequest} req - Request. `req.body` debe contener los datos del producto.
 * @param {ExpressResponse} res - Response.
 * @returns {Promise<void>}
 */
export const crearProducto = async (req, res) => {
  try {
    const nuevoProducto = new Product(req.body);
    await nuevoProducto.save();
    res.status(201).json({
      mensaje: "El producto se creó correctamente",
      producto: nuevoProducto
    });
  } catch (error) {
    console.error(error);
    if (error.code === 11000) {
      return res.status(400).json({
        mensaje: "Ya existe un producto con ese código"
      });
    }
    res.status(500).json({
      mensaje: "Error interno del servidor al crear producto"
    });
  }
};
/**
 * Actualiza un producto por id.
 * @async
 * @param {ExpressRequest} req - Request. `req.params.id` id, `req.body` datos para actualizar.
 * @param {ExpressResponse} res - Response.
 * @returns {Promise<void>}
 */
export const actualizarProducto = async (req, res) => {
  try {
    const productoActualizado = await Product.findByIdAndUpdate(
      req.params.id, 
      req.body, 
      { new: true, runValidators: true }
    );
    
    if (!productoActualizado) {
      return res.status(404).json({
        mensaje: "El producto no fue encontrado"
      });
    }
    
    res.status(200).json({
      mensaje: "El producto se actualizó correctamente",
      producto: productoActualizado
    });
  } catch (error) {
    console.error(error);
    if (error.code === 11000) {
      return res.status(400).json({
        mensaje: "Ya existe un producto con ese código"
      });
    }
    res.status(500).json({
      mensaje: "Error interno del servidor al actualizar producto"
    });
  }
};
/**
 * Elimina un producto por id.
 * @async
 * @param {ExpressRequest} req - Request. `req.params.id` id del producto.
 * @param {ExpressResponse} res - Response.
 * @returns {Promise<void>}
 */
export const eliminarProducto = async (req, res) => {
  try {
    const productoEliminado = await Product.findByIdAndDelete(req.params.id);

      });

      producto: productoEliminado
    });
  } catch (error) {
    console.error(error);
